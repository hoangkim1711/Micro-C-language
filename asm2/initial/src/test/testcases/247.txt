
            int Fibonaci(int i) {
            if (i == 0) { return 0; }
            if(i == 1) { return 1;}
            return Fibonaci(i-1) + Fibonaci(i-2);
        }
        int  main() {
            int i;
            for (i = 0; i < 10; i=i+1) { printf("%f", Fibonaci(i)); }
            printf("===========================");
            return 0;
        }
